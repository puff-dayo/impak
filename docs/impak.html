impak

impak – space-efficient image collection format based on pixel-diff patches.

Quick start

Pack a folder of PNG variants into a single file:

import impak

with impak.create("variants.impak", mode="vs_first") as w:
    for path in sorted(Path("frames/").glob("*.png")):
        w.add(path, name=path.stem)

Read them back:

with impak.open("variants.impak") as r:
    print(r.info())          # human-readable summary
    img = r[0]               # PIL Image, by index
    img = r["frame_01"]      # by name
    for img in r:            # iterate all
        img.show()
Diff modes
Mode Description
lto (default) Each frame searches all prior frames and picks the best reference (smallest patch payload), forming a DAG of deltas. A keyframe is stored when it's smaller than the best delta. Params: lto_candidates (default 6), max_ref_depth (default 8).
vs_first Every frame is a patch against frame 0.
vs_prior Every frame is a patch against the immediately prior frame.
keyframe Like vs_prior but stores a full image every K frames. Param: keyframe_interval (default 10).
manual User-designated baselines are stored first and searched before each content frame. Params: baselines (required), fallback_mode (default vs_prior). 
Example:

    with impak.create(
        "out.impak",
        mode="manual",
        baselines=["day_bg.png", "night_bg.png"],
        fallback_mode="vs_prior",
    ) as w:
        for path in content_frames:
            w.add(path)

 65    Sequential writer for .impak collections.
 66
 67    Parameters
 68    ----------
 69    path              : output file path
 70    mode              : "vs_first" | "vs_prior" | "keyframe" | "lto" | "manual"
 71    keyframe_interval : (keyframe mode only) store a full image every N frames
 72    threshold         : pixel delta considered "changed" (0 = perfectly lossless)
 73    tile_size         : diff grid cell size in pixels
 74    merge_gap         : pixels gap between changed tiles before they are merged
 75    auto_keyframe_sim : if similarity drops below this fraction a keyframe is
 76                        forced regardless of mode (0.0 = never force)
 77    workers           : thread-pool size used for parallel patch compression and
 78                        LTO candidate probing.  None = OS default

 80                        parallelism entirely.

 82                        pinned reference anchors.  They are stored as hidden

 84                        but ImpakReader hides them from callers — only content
 85                        frames are visible when iterating or indexing.
 86                        Baselines are automatically resized to match the canvas
 87                        size established by the first content frame (or the
 88                        first baseline if all are the same size).
 89    fallback_mode     : (manual mode) diff strategy applied when no baseline
 90                        yields a patch set smaller than this alternative.
 91                        Accepts "vs_first", "vs_prior", "keyframe", or "lto".

166    ) -> int:
167        """
168        Add one content image to the collection.  Returns the 0-based frame
169        index as seen by readers (i.e. not counting hidden baseline frames).
170
171        *image* may be a Pillow Image or a file path.
172        *name* is stored in per-frame metadata (useful for later retrieval).
173        *metadata* is an arbitrary JSON-serialisable dict merged with name.
174        """
175        if self._closed:
176            raise RuntimeError("Writer is already closed")
177
178        if isinstance(image, (str, Path)):
179            image = Image.open(image)
180
181        image = image.convert("RGBA")
182
183        if self._canvas is None:
184            self._canvas = (image.width, image.height)
185        elif (image.width, image.height) != self._canvas:
186            raise ValueError(
187                f"Frame {len(self._frames)}: image size {image.size} does not "
188                f"match canvas {self._canvas}"
189            )
190
191        if self._pending_baselines:
192            self._inject_baselines(self._pending_baselines)
193            self._pending_baselines = []
194
195        frame_id = len(self._frames)
196
197        frame_type, ref_id, patches = self._encode_frame(image, frame_id)
198
199        meta: dict = {}
200        if name:

277        Store each baseline as a hidden leading FRAME_KEYFRAME.
278
279        Called from the first add() so self._canvas is already set.
280        Baselines are resized to the canvas if their native size differs.
281        """
282        for idx, raw in enumerate(baselines):
283            if isinstance(raw, (str, Path)):
284                img = Image.open(raw).convert("RGBA")
285            else:
286                img = raw.convert("RGBA")
287
288            # LANCZOS? not sure..
289            if img.size != self._canvas:
290                img = img.resize(self._canvas, Image.LANCZOS)
291
292            frame_id = len(self._frames)
293            cw, ch = self._canvas
294            compressed = _encode_crop(img, 0, 0, cw, ch, codec=self.codec, quality=self.quality)

297            meta_bytes = json.dumps(

320    def _encode_frame(self, image: Image.Image, frame_id: int):
321        """Route to the appropriate encoder. Returns (frame_type, ref_id, patches)."""
322        if self.mode_id == MODE_MANUAL:
323            return self._encode_frame_manual(image, frame_id)
324
325        if frame_id == 0:
326            return self._make_keyframe(image, frame_id)
327
328        if self.mode_id == MODE_LTO:
329            return self._encode_frame_lto(image, frame_id)
330
331        if self.mode_id == MODE_KEYFRAME and (frame_id % self.keyframe_interval == 0):
332            return self._make_keyframe(image, frame_id)
333
334        if self.mode_id == MODE_VS_FIRST:
335            ref_id = 0
336        elif self.mode_id == MODE_VS_PRIOR:
337            ref_id = frame_id - 1
338        else:
339            ref_id = frame_id - 1
340
341        return self._diff_against(image, frame_id, ref_id)

343    def _ref_chain_depth(self, frame_id: int) -> int:
344        depth = 0
345        fid = frame_id
346        while self._frames[fid]["frame_type"] != FRAME_KEYFRAME:
347            fid = self._frames[fid]["ref_frame_id"]
348            depth += 1
349            if depth > len(self._frames):
350                break
351        return depth

353    def _encode_frame_manual(self, image: Image.Image, frame_id: int):
354        """
355        Manual-mode encoder: probe designated baseline frames first, then fall
356        back to the configured fallback mode if no baseline wins.
357
358        Algorithm
359        ---------
360        1. Score all baselines by pixel similarity (numpy, cheap).
361        2. Probe the top-K baselines (full patch compression, parallel).
362        3. Compute the fallback result via _encode_frame_fallback().
363        4. Three-way size comparison: keyframe vs best-baseline vs fallback.
364           Return whichever is smallest.
365        """
366        new_arr = np.array(image.convert("RGBA"), dtype=np.int16)
367
368        def _score_baseline(bid: int):
369            ref_arr = np.array(self._ref_images[bid].convert("RGBA"), dtype=np.int16)
370            diff = np.abs(new_arr - ref_arr).max(axis=2)
371            sim = float((diff <= self.threshold).sum()) / (new_arr.shape[0] * new_arr.shape[1])
372            return sim, bid
373
374        if len(self._baseline_ids) <= 2 or self.workers == 1:
375            scored = [_score_baseline(bid) for bid in self._baseline_ids]
376        else:
377            with ThreadPoolExecutor(max_workers=self.workers) as pool:
378                scored = list(pool.map(_score_baseline, self._baseline_ids))
379
380        scored.sort(key=lambda x: -x[0])
381        top_baselines = [bid for (_, bid) in scored[: self.lto_candidates]]
382
383        def _probe_baseline(bid: int):
384            patches = compute_patches(
385                self._ref_images[bid], image,
386                threshold=self.threshold,
387                tile_size=self.tile_size,
388                merge_gap=self.merge_gap,
389                codec=self.codec,
390                quality=self.quality,
391                workers=1,
392            )
393            return bid, patches, sum(len(p[4]) for p in patches)
394
395        if len(top_baselines) <= 1 or self.workers == 1:
396            baseline_results = [_probe_baseline(bid) for bid in top_baselines]
397        else:
398            pool_size = min(len(top_baselines), self.workers) if self.workers else len(top_baselines)
399            with ThreadPoolExecutor(max_workers=pool_size) as pool:
400                baseline_results = list(pool.map(_probe_baseline, top_baselines))
401
402        best_bl_ref, best_bl_patches, best_bl_size = min(baseline_results, key=lambda r: r[2])
403
404        fb_type, fb_ref_id, fb_patches = self._encode_frame_fallback(image, frame_id)
405        fb_size = sum(len(p[4]) for p in fb_patches)
406
407        iw, ih = image.size
408        kf_size = len(_encode_crop(image, 0, 0, iw, ih, codec=self.codec, quality=self.quality))
409
410        if kf_size <= best_bl_size and kf_size <= fb_size:
411            return self._make_keyframe(image, frame_id)
412        if best_bl_size <= fb_size:
413            return FRAME_DELTA, best_bl_ref, best_bl_patches
414        return fb_type, fb_ref_id, fb_patches

416    def _encode_frame_fallback(self, image: Image.Image, frame_id: int):
417        """
418        Run the configured fallback_mode for this frame.
419
420        content_id is the frame's index within the content-only sequence
421        (i.e. excluding the hidden baseline keyframes at the front).
422        This keeps keyframe_interval counting and vs_first anchoring correct.
423        """
424        content_id = frame_id - self._baseline_count
425
426        if self.fallback_mode_id == MODE_VS_FIRST:
427            if content_id == 0:
428                return self._make_keyframe(image, frame_id)
429            ref_id = self._baseline_count   # first content frame
430            return self._diff_against(image, frame_id, ref_id)
431
432        if self.fallback_mode_id == MODE_VS_PRIOR:
433            if content_id == 0:
434                return self._make_keyframe(image, frame_id)
435            return self._diff_against(image, frame_id, frame_id - 1)
436
437        if self.fallback_mode_id == MODE_KEYFRAME:
438            if content_id == 0 or (content_id % self.keyframe_interval == 0):
439                return self._make_keyframe(image, frame_id)
440            return self._diff_against(image, frame_id, frame_id - 1)
441
442        if self.fallback_mode_id == MODE_LTO:
443            if content_id == 0:
444                return self._make_keyframe(image, frame_id)
445            return self._encode_frame_lto(image, frame_id)
446
447        return self._make_keyframe(image, frame_id)
448
449    def _diff_against(self, image: Image.Image, frame_id: int, ref_id: int):
450        """Compute FRAME_DELTA against ref_id, with auto-keyframe guard."""
451        ref_img = self._ref_images[ref_id]
452
453        if self.auto_keyframe_sim > 0:
454            a = np.array(ref_img.convert("RGBA"), dtype=np.int16)
455            b = np.array(image.convert("RGBA"), dtype=np.int16)
456            diff_arr = np.abs(a - b).max(axis=2)
457            if (diff_arr <= self.threshold).sum() / (a.shape[0] * a.shape[1]) < self.auto_keyframe_sim:
458                return self._make_keyframe(image, frame_id)
459
460        patches = compute_patches(
461            ref_img, image,
462            threshold=self.threshold,
463            tile_size=self.tile_size,
464            merge_gap=self.merge_gap,
465            codec=self.codec,
466            quality=self.quality,
467            workers=self.workers,
468        )
469        return FRAME_DELTA, ref_id, patches

471    def _encode_frame_lto(self, image: Image.Image, frame_id: int):
472        """LTO: find the prior frame yielding the smallest patch set."""
473        new_arr = np.array(image.convert("RGBA"), dtype=np.int16)
474
475        eligible = [
476            cid for cid in range(frame_id)
477            if self._ref_chain_depth(cid) + 1 <= self.max_ref_depth
478        ]
479        if not eligible:
480            return self._make_keyframe(image, frame_id)
481
482        def _score(cid: int):
483            ref_arr = np.array(self._ref_images[cid].convert("RGBA"), dtype=np.int16)
484            diff = np.abs(new_arr - ref_arr).max(axis=2)
485            return float((diff <= self.threshold).sum()) / (new_arr.shape[0] * new_arr.shape[1]), cid
486
487        if len(eligible) <= 2 or self.workers == 1:
488            scored = [_score(cid) for cid in eligible]
489        else:
490            with ThreadPoolExecutor(max_workers=self.workers) as pool:
491                scored = list(pool.map(_score, eligible))
492
493        scored.sort(key=lambda x: -x[0])
494        candidates = [cid for (_, cid) in scored[: self.lto_candidates]]
495
496        def _probe(cid: int):

501                merge_gap=self.merge_gap,
502                codec=self.codec,
503                quality=self.quality,
504                workers=1,
505            )
506            return cid, patches, sum(len(p[4]) for p in patches)
507
508        if len(candidates) <= 1 or self.workers == 1:
509            results = [_probe(cid) for cid in candidates]
510        else:
511            pool_size = min(len(candidates), self.workers) if self.workers else len(candidates)
512            with ThreadPoolExecutor(max_workers=pool_size) as pool:
513                results = list(pool.map(_probe, candidates))
514
515        best_ref_id, best_patches, best_size = min(results, key=lambda r: r[2])
516
517        iw, ih = image.size
518        kf_data = _encode_crop(image, 0, 0, iw, ih, codec=self.codec, quality=self.quality)
519        if len(kf_data) <= best_size:
520            return self._make_keyframe(image, frame_id)

In manual-mode files the leading baseline keyframes are automatically detected (via their {"_baseline": true} metadata) and excluded from all public-facing operations. They are still used internally when decoding delta frames that reference them.

 79    tile_size: int = 32,

 85    """
 86    Compare *new_img* against *ref_img* using a tile grid.
 87
 88    Parameters
 89    ----------
 90    ref_img    : PIL Image – reference (what we're diffing against)
 91    new_img    : PIL Image – the new frame
 92    threshold  : per-channel absolute pixel delta that counts as "changed"
 93                 (use 0 for perfectly lossless; 3-6 absorbs JPEG noise)
 94    tile_size  : grid cell size in pixels (smaller = finer granularity but
 95                 more patch overhead; 16–64 is a good range)
 96    merge_gap  : changed tiles within this many pixels of each other on the
 97                 same row are merged into one patch (reduces patch count)
 98    workers    : thread-pool size for parallel patch compression.
 99                 None = use ThreadPoolExecutor default (cpu_count × 4 or so).
100                 Set to 1 to disable parallelism entirely.
101
102    Returns
103    -------
104    List of (x, y, w, h, compressed_bytes) tuples.
105    Empty list if the images are identical within threshold.
106    """
107    ref = np.array(ref_img.convert("RGBA"), dtype=np.int16)
108    new = np.array(new_img.convert("RGBA"), dtype=np.int16)
109
110    if ref.shape != new.shape:
111        raise ValueError(
112            f"Image dimensions differ: ref={ref.shape[:2]} new={new.shape[:2]}"
113        )
114
115    h, w = ref.shape[:2]
116    diff = np.abs(new - ref).max(axis=2)
117
118    cols = (w + tile_size - 1) // tile_size
119    rows = (h + tile_size - 1) // tile_size
120
121    pad_h = rows * tile_size - h
122    pad_w = cols * tile_size - w
123    if pad_h or pad_w:
124        diff_padded = np.pad(diff, ((0, pad_h), (0, pad_w)), constant_values=0)
125    else:
126        diff_padded = diff
127
128    tile_max = (
129        diff_padded
130        .reshape(rows, tile_size, cols, tile_size)
131        .max(axis=(1, 3))
132    )
133    changed_mask = tile_max > threshold
134    tile_rows, tile_cols = np.where(changed_mask)
135    changed_tiles = list(zip(tile_rows.tolist(), tile_cols.tolist()))
136
137    if not changed_tiles:
138        return []
139
140    rects = _tiles_to_rects(changed_tiles, tile_size, w, h)
141    merged = _merge_rects(rects, merge_gap, tile_size, w, h)
142
143    if len(merged) <= 2 or workers == 1:
144        patches: List[Patch] = [
145            (rx, ry, rw, rh, _encode_crop(new_img, rx, ry, rw, rh, codec=codec, quality=quality))
146            for (rx, ry, rw, rh) in merged
147        ]
148    else:
149        def _compress_rect(rect: tuple) -> Patch:
150            rx, ry, rw, rh = rect
151            return rx, ry, rw, rh, _encode_crop(new_img, rx, ry, rw, rh, codec=codec, quality=quality)
152
153        with ThreadPoolExecutor(max_workers=workers) as pool:
154            patches = list(pool.map(_compress_rect, merged))
155
156    return patches

def similarity_score(img_a: PIL.Image.Image, img_b: PIL.Image.Image) -> float: 
253def similarity_score(img_a: Image.Image, img_b: Image.Image) -> float:
254    """
255    Return fraction of pixels that are identical (0.0 = completely different,
256    1.0 = identical).  Useful for deciding whether to force a keyframe.
257    """
258    a = np.array(img_a.convert("RGBA"), dtype=np.int16)
259    b = np.array(img_b.convert("RGBA"), dtype=np.int16)
260    if a.shape != b.shape:
261        return 0.0
262    diff = np.abs(a - b).max(axis=2)
263    identical_pixels = int((diff == 0).sum())
264    total = a.shape[0] * a.shape[1]
265    return identical_pixels / total
